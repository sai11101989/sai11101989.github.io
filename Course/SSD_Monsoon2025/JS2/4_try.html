JavaScript Error Handling: try-catch, finally, throw, Optional Catch Binding

1. try-catch

try {
  // Code that may throw error
  throw new Error("An error occurred!");
} catch (e) {
  console.log("Caught error: " + e.message);
}

2. try-catch-finally

try {
  throw new Error("Another error!");
} catch (e) {
  console.log("Caught: " + e.message);
} finally {
  console.log("Finally block always runs.");
}

3. throw

function checkNumber(num) {
  if (typeof num !== "number") {
    throw "Not a number!";
  }
  return num;
}

4. Optional Catch Binding

try {
  throw new Error("Error without catch parameter!");
} catch {
  console.log("Error caught with no catch parameter.");
}
