JavaScript Iterators, Generators, yield, Symbol.iterator, Array Iterator

1. Iterator Example

const myIterator = {
  data: [10, 20, 30],
  index: 0,
  next() {
    if (this.index < this.data.length) {
      return { value: this.data[this.index++], done: false };
    } else {
      return { done: true };
    }
  }
};

console.log(myIterator.next()); // { value: 10, done: false }

2. Generator & yield Example

function* generateNumbers() {
  yield 1;
  yield 2;
  yield 3;
}
const gen = generateNumbers();

console.log(gen.next()); // { value: 1, done: false }

3. Symbol.iterator Example

const myIterable = {
  data: [100, 200, 300],
  [Symbol.iterator]() {
    let index = 0;
    const data = this.data;
    return {
      next() {
        if (index < data.length) {
          return { value: data[index++], done: false };
        } else {
          return { done: true };
        }
      }
    };
  }
};

for(let val of myIterable) {
  console.log(val);
}

4. Array Iterator Example

const arr = ['a', 'b', 'c'];
const arrIterator = arr[Symbol.iterator]();

console.log(arrIterator.next()); // { value: 'a', done: false }
